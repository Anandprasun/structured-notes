Use the prompt as the system message and your notes as the user message. Works with the Anthropic, OpenAI, or Google SDKs.

      
SYSTEM = "<the prompt above>"
messages = [
  {"role": "system", "content": SYSTEM},   # Anthropic: pass as system=SYSTEM
  {"role": "user",   "content": raw_notes},
]
